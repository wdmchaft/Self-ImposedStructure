tell application "iCal"	tell calendar "<calName>"		set endDate to date "<endDate>"		set dueList to every todo	end tell	set recList to {}	repeat with td in dueList		set compDate to completion date of td		set dueDate to due date of td		if (compDate is missing value) then			if (dueDate is missing value) then				set prioTxt to priority of td				set descTxt to description of td				set idStr to uid of td				set dueStr to due date of td				set sumTxt to summary of td				set myRecord to {sum:sumTxt, pri:prioTxt, desc:descTxt, unique:idStr, due:dueStr}				copy myRecord to the end of recList			else if (dueDate < endDate) then				set prioTxt to priority of td				set descTxt to description of td				set idStr to uid of td				set dueStr to due date of td				set sumTxt to summary of td				set myRecord to {sum:sumTxt, pri:prioTxt, desc:descTxt, unique:idStr, due:dueStr}				copy myRecord to the end of recList			end if		end if	end repeat	return recListend tell
